52 if (dot1 == StringPiece::npos) {
53 throw std::invalid_argument("could not find first dot");
54 }
55 auto v1 = folly::to<int>(release.subpiece(0, dot1));
56
57 auto dot2 = release.find('.', dot1 + 1);
