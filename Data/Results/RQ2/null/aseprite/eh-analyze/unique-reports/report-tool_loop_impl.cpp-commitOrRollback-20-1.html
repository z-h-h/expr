686 const bool selectTiles)
687{
688 ToolLoopParams params;
689 params.tool = editor->getCurrentEditorTool();
690 params.ink = editor->getCurrentEditorInk();
691 if (!params.tool || !params.ink)
692 return nullptr;
693
694 if (selectTiles) {
695 params.tool = App::instance()->toolBox()->getToolById(tools::WellKnownTools::RectangularMarquee);
696 params.ink = params.tool->getInk(button == tools::Pointer::Left ? 0: 1);
697 }
698
699 Site site = editor->getSite();
700
701 // For selection tools, we can use any layer (even without layers at
702 // all), so we specify a nullptr here as the active layer. This is
703 // used as a special case by the render::Render class to show the
704 // preview image/selection stroke as a xor'd overlay in the render
705 // result.
706 //
707 // Anyway this cannot be used in 'magic wand' tool (isSelection +
708 // isFloodFill) because we need the original layer source
709 // image/pixels to stop the flood-fill algorithm.
710 if (params.ink->isSelection() &&
711 !params.tool->getPointShape(
712 button != tools::Pointer::Left ? 1: 0)->isFloodFill()) {
713 site.layer(nullptr);
714 }
715 else {
716 Layer* layer = site.layer();
717 if (!layer) {
718 StatusBar::instance()->showTip(
719 1000, "There is no active layer");
720 return nullptr;
721 }
722 else if (!layer->isVisibleHierarchy()) {
723 StatusBar::instance()->showTip(
724 1000, fmt::format("Layer '{}' is hidden", layer->name()));
725 return nullptr;
726 }
727 // If the active layer is read-only.
728 else if (layer_is_locked(editor)) {
729 return nullptr;
730 }
731 // If the active layer is reference.
732 else if (layer->isReference()) {
733 StatusBar::instance()->showTip(
734 1000, fmt::format("Layer '{}' is reference, cannot be modified", layer->name()));
735 return nullptr;
736 }
737 }
738
739 // Get fg/bg colors
740 ColorBar* colorbar = ColorBar::instance();
741 params.fg = colorbar->getFgColor();
742 params.bg = colorbar->getBgColor();
743
744 if (!params.fg.isValid() ||
745 !params.bg.isValid()) {
746 if (Preferences::instance().colorBar.showInvalidFgBgColorAlert()) {
747 OptionalAlert::show(
748 Preferences::instance().colorBar.showInvalidFgBgColorAlert,
749 1, Strings::alerts_invalid_fg_or_bg_colors());
750 return nullptr;
751 }
752 }
753
754 // Create the new tool loop
755 try {
756 params.button =
757 (button == tools::Pointer::Left ? tools::ToolLoop::Left:
758 tools::ToolLoop::Right);
759
760 params.controller =
761 (convertLineToFreehand ?
762 App::instance()->toolBox()->getControllerById(
763 tools::WellKnownControllers::LineFreehand):
764 params.tool->getController(params.button));
765
766 const bool saveLastPoint =
767 (params.ink->isPaint() &&
768 (params.controller->isFreehand() ||
769 convertLineToFreehand));
770
771 params.brush = App::instance()->contextBar()
772 ->activeBrush(params.tool, params.ink);
773

780 if (selectTiles)
781 toolLoop->forceSnapToTiles();

784 }
