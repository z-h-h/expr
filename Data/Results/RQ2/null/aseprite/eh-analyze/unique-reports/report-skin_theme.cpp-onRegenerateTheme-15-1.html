728 xml_filename.c_str());
729 }
730 }

733 style->addLayer(layer);
734
735 xmlLayer = xmlLayer->NextSiblingElement();
736 }
737
738 xmlStyle = xmlStyle->NextSiblingElement();

743 backward->createMissingStyles(this);

1108{
1109 // Draw the text
1110 gfx::Rect bounds = widget->getEntryTextBounds();
1111

1113 int scroll = delegate.index();
1114
1115 const std::string& textString = widget->text();
1116 base::utf8_const_iterator utf8_it((textString.begin()));
1117 int textlen = base::utf8_length(textString);
1118 scroll = std::min(scroll, textlen);
1119 if (scroll)
1120 utf8_it += scroll;
1121
1122 g->drawText(utf8_it,
1123 base::utf8_const_iterator(textString.end()),
1124 colors.text(), ColorNone,
1125 bounds.origin(), &delegate);
1126
1127 bounds.x += delegate.textBounds().w;
1128
1129 // Draw suffix if there is enough space
1130 if (!widget->getSuffix().empty()) {
1131 Rect sufBounds(bounds.x, bounds.y,
1132 bounds.x2()-widget->childSpacing()*guiscale()-bounds.x,
1133 widget->textHeight());

1144 if (!delegate.caretDrawn()) {
1145 gfx::Rect charBounds(bounds.x+widget->bounds().x,

1148 delegate.postDrawChar(charBounds);

1271void SkinTheme::paintSlider(PaintEvent& ev)
1272{
1273 Graphics* g = ev.graphics();
1274 Slider* widget = static_cast<Slider*>(ev.getSource());
1275 const Rect bounds = widget->clientBounds();
1276 int min, max, value;
1277
1278 // Outside borders
1279 gfx::Color bgcolor = widget->bgColor();
1280 if (!is_transparent(bgcolor))
1281 g->fillRect(bgcolor, bounds);
1282
1283 widget->getSliderThemeInfo(&min, &max, &value);
1284
1285 Rect rc = bounds;
1286 rc.shrink(widget->border());
1287 int x;
1288 if (min != max)
1289 x = rc.x + rc.w * (value-min) / (max-min);
1290 else
1291 x = rc.x;
1292
1293 rc = bounds;
1294
1295 // The mini-look is used for sliders with tiny borders.
1296 bool isMiniLook = false;
1297
1298 // The BG painter is used for sliders without a number-indicator and
1299 // customized background (e.g. RGB sliders)
1300 ISliderBgPainter* bgPainter = NULL__null;
1301
1302 const auto skinPropery = std::static_pointer_cast<SkinProperty>(widget->getProperty(SkinProperty::Name));
1303 if (skinPropery)
1304 isMiniLook = (skinPropery->getLook() == MiniLook);
1305
1306 const auto skinSliderPropery = std::static_pointer_cast<SkinSliderProperty>(widget->getProperty(SkinSliderProperty::Name));
1307 if (skinSliderPropery)
1308 bgPainter = skinSliderPropery->getBgPainter();
1309
1310 // Draw customized background
1311 if (bgPainter) {
1312 SkinPartPtr nw = parts.miniSliderEmpty();
1313 os::Surface* thumb =
1314 (widget->hasFocus() ? parts.miniSliderThumbFocused()->bitmap(0):
1315 parts.miniSliderThumb()->bitmap(0));
1316
1317 // Draw background
1318 g->fillRect(BGCOLOR(getWidgetBgColor(widget)), rc);
1319
1320 // Draw thumb
1321 int thumb_y = rc.y;
1322 if (rc.h > thumb->height()*3)
1323 rc.shrink(Border(0, thumb->height(), 0, 0));
1324
1325 // Draw borders
1326 if (rc.h > 4*guiscale()) {
1327 rc.shrink(Border(3, 0, 3, 1) * guiscale());
1328 drawRect(g, rc, nw.get());
1329 }
1330
1331 // Draw background (using the customized ISliderBgPainter implementation)
1332 rc.shrink(Border(1, 1, 1, 2) * guiscale());
1333 if (!rc.isEmpty())
1334 bgPainter->paint(widget, g, rc);
1335
1336 g->drawRgbaSurface(thumb, x-thumb->width()/2, thumb_y);
1337 }
1338 else {
1339 // Draw borders
1340 SkinPartPtr full_part;
1341 SkinPartPtr empty_part;
1342
1343 if (isMiniLook) {
1344 full_part = widget->hasMouseOver() ? parts.miniSliderFullFocused():
1345 parts.miniSliderFull();
1346 empty_part = widget->hasMouseOver() ? parts.miniSliderEmptyFocused():
1347 parts.miniSliderEmpty();
1348 }
1349 else {
1350 full_part = widget->hasFocus() ? parts.sliderFullFocused():
1351 parts.sliderFull();
1352 empty_part = widget->hasFocus() ? parts.sliderEmptyFocused():
1353 parts.sliderEmpty();
1354 }
1355
1356 if (value == min)
1357 drawRect(g, rc, empty_part.get());
1358 else if (value == max)
1359 drawRect(g, rc, full_part.get());
1360 else
1361 drawRect2(g, rc, x,
1362 full_part.get(), empty_part.get());
1363
1364 // Draw text
1365 std::string old_text = widget->text();
1366 widget->setTextQuiet(widget->convertValueToText(value));
1367
1368 gfx::Rect textrc;
1369 int textAlign;
1370 calcTextInfo(widget, widget->style(), bounds, textrc, textAlign);
1371
1372 {
1373 IntersectClip clip(g, Rect(rc.x, rc.y, x-rc.x+1, rc.h));
1374 if (clip) {
1375 drawText(g, nullptr,
1376 colors.sliderFullText(), ColorNone,
1377 widget, textrc, textAlign, widget->mnemonic());
1378 }
1379 }
1380
1381 {
1382 IntersectClip clip(g, Rect(x+1, rc.y, rc.w-(x-rc.x+1), rc.h));
1383 if (clip) {
1384 drawText(g, nullptr,
1385 colors.sliderEmptyText(),
1386 ColorNone, widget, textrc, textAlign, widget->mnemonic());
1387 }
1388 }
1389
1390 widget->setTextQuiet(old_text.c_str());
1391 }
1392}

1444void SkinTheme::drawText(Graphics* g, const char* t,
1445 const gfx::Color fgColor,
1446 const gfx::Color bgColor,
1447 const Widget* widget,
1448 const Rect& rc,
1449 const int textAlign,
1450 const int mnemonic)
1451{
1452 if (t || widget->hasText()) {
1453 Rect textrc;
1454
1455 g->setFont(widget->font());
1456
1457 if (!t)
1458 t = widget->text().c_str();
1459
1460 textrc.setSize(g->measureUIText(t));
1461
1462 // Horizontally text alignment
1463
1464 if (textAlign & RIGHT)
1465 textrc.x = rc.x + rc.w - textrc.w - 1;
1466 else if (textAlign & CENTER)
1467 textrc.x = rc.center().x - textrc.w/2;

1469 textrc.x = rc.x;

1472

1478 textrc.y = rc.y;

1481 if (!is_transparent(bgColor)) {
1482 if (!widget->isEnabled())
1483 g->fillRect(bgColor, Rect(textrc).inflate(guiscale(), guiscale()));
1484 else
1485 g->fillRect(bgColor, textrc);
1486 }
1487
1488 // Text
1489 Rect textWrap = textrc.createIntersection(
1490 // TODO add ui::Widget::getPadding() property
1491 // Rect(widget->clientBounds()).shrink(widget->border()));
1492 widget->clientBounds()).inflate(0, 1*guiscale());
1493
1494 IntersectClip clip(g, textWrap);
1495 if (clip) {
1496 if (!widget->isEnabled()) {
1497 // Draw white part
1498 g->drawUIText(
1499 t,
1500 colors.background(),
1501 gfx::ColorNone,
1502 textrc.origin() + Point(guiscale(), guiscale()),
1503 mnemonic);
1504 }
1505
1506 g->drawUIText(
1507 t,
1508 (!widget->isEnabled() ?
1509 colors.disabled():
1510 (gfx::geta(fgColor) > 0 ? fgColor :
1511 colors.text())),
1512 bgColor, textrc.origin(),
1513 mnemonic);
1514 }
