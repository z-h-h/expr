573bool SSLContext::setRandomizedAdvertisedNextProtocols(
574 const std::list<NextProtocolsItem>& items) {
575 unsetNextProtocols();
576 if (items.empty()) {
577 return false;
578 }
579 int total_weight = 0;
580 for (const auto& item : items) {
581 if (item.protocols.empty()) {
582 continue;
583 }
584 AdvertisedNextProtocolsItem advertised_item;
585 advertised_item.length = 0;
586 for (const auto& proto : item.protocols) {
587 ++advertised_item.length;
588 auto protoLength = proto.length();
589 if (protoLength >= 256) {
590 deleteNextProtocolsStrings();
591 return false;
592 }
593 advertised_item.length += unsigned(protoLength);
594 }
595 advertised_item.protocols = new unsigned char[advertised_item.length];
596 if (!advertised_item.protocols) {
597 throw std::runtime_error("alloc failure");
598 }
599 unsigned char* dst = advertised_item.protocols;
600 for (auto& proto : item.protocols) {
601 auto protoLength = uint8_t(proto.length());
602 *dst++ = (unsigned char)protoLength;
603 memcpy(dst, proto.data(), protoLength);
604 dst += protoLength;
605 }
606 total_weight += item.weight;
607 advertisedNextProtocols_.push_back(advertised_item);
608 advertisedNextProtocolWeights_.push_back(item.weight);
609 }
610 if (total_weight == 0) {
611 deleteNextProtocolsStrings();
612 return false;
613 }
614 nextProtocolDistribution_ = std::discrete_distribution<>(
615 advertisedNextProtocolWeights_.begin(),
616 advertisedNextProtocolWeights_.end());
617 SSL_CTX_set_alpn_select_cb(ctx_, alpnSelectCallback, this);
618 // Client cannot really use randomized alpn
619 // Note that this function reverses the typical return value convention
620 // of openssl and returns 0 on success.
621 return SSL_CTX_set_alpn_protos(
622 ctx_,
623 advertisedNextProtocols_[0].protocols,
624 advertisedNextProtocols_[0].length) == 0;
625}

628 for (auto protocols : advertisedNextProtocols_) {
629 delete[] protocols.protocols;
630 }
631 advertisedNextProtocols_.clear();
632 advertisedNextProtocolWeights_.clear();
633}

668/**
669 * Match a name with a pattern. The pattern may include wildcard. A single
670 * wildcard "*" can match up to one component in the domain name.
671 *
672 * @param host Host name, typically the name of the remote host
673 * @param pattern Name retrieved from certificate
674 * @param size Size of "pattern"
675 * @return True, if "host" matches "pattern". False otherwise.
676 */
677bool SSLContext::matchName(const char* host, const char* pattern, int size) {
678 bool match = false;
679 int i = 0, j = 0;
680 while (i < size && host[j] != '\0') {
681 if (toupper(pattern[i]) == toupper(host[j])) {
682 i++;
683 j++;
684 continue;
685 }
686 if (pattern[i] == '*') {
687 while (host[j] != '.' && host[j] != '\0') {
688 j++;
689 }
690 i++;
691 continue;
692 }
693 break;
694 }
695 if (i == size && host[j] == '\0') {
696 match = true;
697 }
698 return match;
699}

731std::string SSLContext::getErrors(int errnoCopy) {
732 std::string errors;
733 unsigned long errorCode;
734 char message[256];
735
736 errors.reserve(512);
737 while ((errorCode = ERR_get_error()) != 0) {
738 if (!errors.empty()) {
739 errors += "; ";
740 }
741 const char* reason = ERR_reason_error_string(errorCode);
742 if (reason == nullptr) {
743 snprintf(message, sizeof(message) - 1, "SSL error # %08lX", errorCode);
744 reason = message;
745 }
746 errors += reason;
747 }
748 if (errors.empty()) {
749 errors = "error code: " + folly::to<std::string>(errnoCopy);
750 }
751 return errors;
752}

766void SSLContext::setupCtx(SSL_CTX* ctx) {
767 // 1) folly::AsyncSSLSocket wants to unconditionally store a client
768 // session, so that is possible to later perform TLS resumption.
769 // For that, we need SSL_SESS_CACHE_CLIENT.
770 //
771 // 2) wangle::SSLSessionCacheManager needs to be able to receive
772 // SSL_SESSIONs that are established through a successful
773 // connection. For that, we need SSL_SESS_CACHE_SERVER. Consequently,
774 // given the requirements of (1), we opt to use SSL_SESS_CACHE_BOTH
775 //
776 // 3) We explicitly disable the OpenSSL internal session cache, as there
777 // is very little we can do to control the memory usage of the internal
778 // session cache. Server side session-id based caching should be explicitly
779 // opted-in by the user, by forcing them to provide an implementation of
780 // a SessionCache interface (e.g. wangle::SSLSessionCacheManager); i.e.,
781 // the user must be cognizant of the fact that doing so would result in
782 // increased memory usage.
783 SSL_CTX_set_session_cache_mode(SSL_CTX_ctrl(ctx,44,(0x0001|0x0002) | (0x0100|0x0200) | 0x0080
,__null)
784 ctx,SSL_CTX_ctrl(ctx,44,(0x0001|0x0002) | (0x0100|0x0200) | 0x0080
,__null)
785 SSL_SESS_CACHE_BOTH | SSL_SESS_CACHE_NO_INTERNAL |SSL_CTX_ctrl(ctx,44,(0x0001|0x0002) | (0x0100|0x0200) | 0x0080
,__null)
786 SSL_SESS_CACHE_NO_AUTO_CLEAR)SSL_CTX_ctrl(ctx,44,(0x0001|0x0002) | (0x0100|0x0200) | 0x0080
,__null);
787
788 SSL_CTX_set_ex_data(ctx, getExDataIndex(), this);
789 SSL_CTX_sess_set_new_cb(ctx, SSLContext::newSessionCallback);
790}

796int SSLContext::newSessionCallback(SSL* ssl, SSL_SESSION* session) {
797 SSL_CTX* ctx = SSL_get_SSL_CTX(ssl);
798 SSLContext* context = getFromSSLCtx(ctx);

802 SSL_SESSION_up_ref(session);

804 cb->onNewSession(ssl, std::move(sessionPtr));

807 // Session will either be moved to session manager or
808 // freed when the unique_ptr goes out of scope
809 auto sessionPtr = folly::ssl::SSLSessionUniquePtr(session);
810 auto sessionManager = folly::ssl::SSLSessionManager::getFromSSL(ssl);
811 if (sessionManager) {
812 sessionManager->onNewSession(std::move(sessionPtr));
813 }
814
815 return 1;
