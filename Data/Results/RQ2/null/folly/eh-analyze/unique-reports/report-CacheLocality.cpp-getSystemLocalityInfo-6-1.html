80// Each level of cache has sharing sets, which are the set of cpus
81// that share a common cache at that level. These are available in a
82// hex bitset form (/sys/devices/system/cpu/cpu0/index0/shared_cpu_map,
83// for example). They are also available in a human-readable list form,
84// as in /sys/devices/system/cpu/cpu0/index0/shared_cpu_list. The list
85// is a comma-separated list of numbers and ranges, where the ranges are
86// a pair of decimal numbers separated by a '-'.
87//
88// To sort the cpus for optimum locality we don't really need to parse
89// the sharing sets, we just need a unique representative from the
90// equivalence class. The smallest value works fine, and happens to be
91// the first decimal number in the file. We load all of the equivalence
92// class information from all of the cpu*/index* directories, order the
93// cpus first by increasing last-level cache equivalence class, then by
94// the smaller caches. Finally, we break ties with the cpu number itself.
95
96/// Returns the first decimal number in the string, or throws an exception
97/// if the string does not start with a number terminated by ',', '-',
98/// '\n', or eos.
99static size_t parseLeadingNumber(const std::string& line) {
100 auto raw = line.c_str();
101 char* end;
102 unsigned long val = strtoul(raw, &end, 10);
103 if (end == raw || (*end != ',' && *end != '-' && *end != '\n' && *end != 0)) {
104 throw std::runtime_error(
105 to<std::string>("error parsing list '", line, "'").c_str());
106 }
107 return val;
108}

111 const std::function<std::string(std::string)>& mapping) {
112 // number of equivalence classes per level
113 std::vector<size_t> numCachesByLevel;

115 // the list of cache equivalence classes, where equivalance classes
116 // are named by the smallest cpu in the class
117 std::vector<std::vector<size_t>> equivClassesByCpu;
118
119 std::vector<size_t> cpus;
120

122 auto cpu = cpus.size();

124 for (size_t index = 0;; ++index) {
125 auto dir = fmt::format(
126 "/sys/devices/system/cpu/cpu{}/cache/index{}/", cpu, index);
127 auto cacheType = mapping(dir + "type");
128 auto equivStr = mapping(dir + "shared_cpu_list");
129 if (cacheType.empty() || equivStr.empty()) {
130 // no more caches
131 break;
132 }
133 if (cacheType[0] == 'I') {
134 // cacheType in { "Data", "Instruction", "Unified" }. skip icache
135 continue;
136 }
137 auto equiv = parseLeadingNumber(equivStr);
138 auto level = levels.size();
139 levels.push_back(equiv);
140
141 if (equiv == cpu) {
142 // we only want to count the equiv classes once, so we do it when
143 // we first encounter them
144 while (numCachesByLevel.size() <= level) {
145 numCachesByLevel.push_back(0);
146 }
147 numCachesByLevel[level]++;
148 }
149 }
150
151 if (levels.empty()) {
152 // no levels at all for this cpu, we must be done
153 break;
154 }
155 equivClassesByCpu.emplace_back(std::move(levels));
156 cpus.push_back(cpu);
157 }
158
159 if (cpus.empty()) {
160 throw std::runtime_error("unable to load cache sharing info");
161 }
162
163 std::sort(cpus.begin(), cpus.end(), [&](size_t lhs, size_t rhs) -> bool {
164 // sort first by equiv class of cache with highest index,
165 // direction doesn't matter. If different cpus have
166 // different numbers of caches then this code might produce
167 // a sub-optimal ordering, but it won't crash
168 auto& lhsEquiv = equivClassesByCpu[lhs];
169 auto& rhsEquiv = equivClassesByCpu[rhs];
170 for (ssize_t i = ssize_t(std::min(lhsEquiv.size(), rhsEquiv.size())) - 1;
171 i >= 0;
172 --i) {
173 auto idx = size_t(i);
174 if (lhsEquiv[idx] != rhsEquiv[idx]) {
175 return lhsEquiv[idx] < rhsEquiv[idx];
176 }
177 }
178
179 // break ties deterministically by cpu
180 return lhs < rhs;
181 });
182
183 // the cpus are now sorted by locality, with neighboring entries closer
184 // to each other than entries that are far away. For striping we want
185 // the inverse map, since we are starting with the cpu
186 std::vector<size_t> indexes(cpus.size());
187 for (size_t i = 0; i < cpus.size(); ++i) {
188 indexes[cpus[i]] = i;
189 }
190
191 return CacheLocality{
192 cpus.size(), std::move(numCachesByLevel), std::move(indexes)};
193}

229 // "processor" is the top line of each record, so when we hit it in
230 // the reverse order then we can emit a record.

235 } else if (line.find("processor") == 0) {

238 cpus.emplace_back(physicalId, coreId, cpu);
239 }
240 }

242 if (cpus.empty()) {
243 throw std::runtime_error("no CPUs parsed from /proc/cpuinfo");
244 }
245 if (maxCpu != cpus.size() - 1) {
246 throw std::runtime_error(
247 "offline CPUs not supported for /proc/cpuinfo cache locality source");
248 }
249
250 std::sort(cpus.begin(), cpus.end());
251 size_t cpusPerCore = 1;
252 while (cpusPerCore < cpus.size() &&
253 std::get<0>(cpus[cpusPerCore]) == std::get<0>(cpus[0]) &&
254 std::get<1>(cpus[cpusPerCore]) == std::get<1>(cpus[0])) {
255 ++cpusPerCore;
256 }
257
258 // we can't tell the real cache hierarchy from /proc/cpuinfo, but it
259 // works well enough to assume there are 3 levels, L1 and L2 per-core
260 // and L3 per socket
261 std::vector<size_t> numCachesByLevel;
262 numCachesByLevel.push_back(cpus.size() / cpusPerCore);
263 numCachesByLevel.push_back(cpus.size() / cpusPerCore);
264 numCachesByLevel.push_back(std::get<0>(cpus.back()) + 1);
265
266 std::vector<size_t> indexes(cpus.size());
267 for (size_t i = 0; i < cpus.size(); ++i) {
268 indexes[std::get<2>(cpus[i])] = i;
269 }
270
271 return CacheLocality{
272 cpus.size(), std::move(numCachesByLevel), std::move(indexes)};
273}
