561// Before executing any command, we drop the pixels (go back to standby).
562void MovingPixelsState::onBeforeCommandExecution(CommandExecutionEvent& ev)
563{
564 Command* command = ev.command();
565
566 MOVPIXS_TRACE("MOVPIXS: onBeforeCommandExecution %s\n", command->id().c_str());
567
568 // If the command is for other editor, we don't drop pixels.
569 if (!isActiveEditor())
570 return;
571
572 if (layer_is_locked(m_editor) &&
573 (command->id() == CommandId::Flip() ||
574 command->id() == CommandId::Cut() ||
575 command->id() == CommandId::Clear() ||
576 command->id() == CommandId::Rotate())) {
577 ev.cancel();
578 return;
579 }
580
581 // We don't need to drop the pixels if a MoveMaskCommand of Content is executed.
582 if (MoveMaskCommand* moveMaskCmd = dynamic_cast<MoveMaskCommand*>(command)) {
583 if (moveMaskCmd->getTarget() == MoveMaskCommand::Content) {
584 if (layer_is_locked(m_editor)) {
585 ev.cancel();
586 return;
587 }
588 gfx::Point delta = moveMaskCmd->getMoveThing().getDelta(UIContext::instance());
589 // Verify Shift condition of the MoveMaskCommand (i.e. wrap = true)
590 if (moveMaskCmd->isWrap()) {
591 m_pixelsMovement->shift(delta.x, delta.y);
592 }
593 else {
594 translate(delta);
595 }
596 // We've processed the selection content movement right here.
597 ev.cancel();
598 return;
599 }
600 }
601 // Don't drop pixels if the user zooms/scrolls/picks a color
602 // using commands.
603 else if ((command->id() == CommandId::Zoom()) ||
604 (command->id() == CommandId::Scroll()) ||
605 (command->id() == CommandId::Eyedropper()) ||
606 // DiscardBrush is used by Eyedropper command
607 (command->id() == CommandId::DiscardBrush())) {
608 // Do not drop pixels
609 return;
610 }
611 // Intercept the "Cut" or "Copy" command to handle them locally
612 // with the current m_pixelsMovement data.
613 else if (command->id() == CommandId::Cut() ||
614 command->id() == CommandId::Copy() ||
615 command->id() == CommandId::Clear()) {
616 // Copy the floating image to the clipboard on Cut/Copy.
617 if (command->id() != CommandId::Clear()) {
618 Doc* document = m_editor->document();
619 std::unique_ptr<Image> floatingImage;
620 std::unique_ptr<Mask> floatingMask;
621 m_pixelsMovement->getDraggedImageCopy(floatingImage, floatingMask);
622
623 clipboard::copy_image(floatingImage.get(),
624 floatingMask.get(),
625 document->sprite()->palette(m_editor->frame()));
626 }
627
628 // Clear floating pixels on Cut/Clear.
629 if (command->id() != CommandId::Copy()) {
630 m_pixelsMovement->trim();
631
632 // Should we keep the mask after an Edit > Clear command?
633 auto keepMask = PixelsMovement::DontKeepMask;
634 if (command->id() == CommandId::Clear() &&
635 Preferences::instance().selection.keepSelectionAfterClear()) {
636 keepMask = PixelsMovement::KeepMask;
637 }
638
639 // Discard the dragged image.
640 m_pixelsMovement->discardImage(PixelsMovement::CommitChanges, keepMask);
641 m_discarded = true;
642
643 // Quit from MovingPixelsState, back to standby.
644 m_editor->backToPreviousState();
645 }
646
647 // Cancel the command, we've simulated it.
648 ev.cancel();
649 return;
650 }
651 // Flip Horizontally/Vertically commands are handled manually to
652 // avoid dropping the floating region of pixels.
653 else if (command->id() == CommandId::Flip()) {
654 if (FlipCommand* flipCommand = dynamic_cast<FlipCommand*>(command)) {
655 this->flip(flipCommand->getFlipType());
656
657 ev.cancel();
658 return;
659 }
660 }
661 // Rotate is quite simple, we can add the angle to the current transformation.
662 else if (command->id() == CommandId::Rotate()) {
663 if (RotateCommand* rotate = dynamic_cast<RotateCommand*>(command)) {
664 if (rotate->flipMask()) {
665 this->rotate(rotate->angle());
666
667 ev.cancel();
668 return;
669 }
670 }
671 }
672 // We can use previous/next frames while transforming the selection
673 // to switch between frames
674 else if (command->id() == CommandId::GotoPreviousFrame() ||
675 command->id() == CommandId::GotoPreviousFrameWithSameTag()) {
676 if (m_pixelsMovement->gotoFrame(-1)) {
677 ev.cancel();
678 return;
679 }
680 }
681 else if (command->id() == CommandId::GotoNextFrame() ||
682 command->id() == CommandId::GotoNextFrameWithSameTag()) {
683 if (m_pixelsMovement->gotoFrame(+1)) {
684 ev.cancel();
685 return;
686 }
687 }
688
689 if (m_pixelsMovement)
690 dropPixels();
691}

453 m_initialData.angle()
454 + atan2((double)(-pos.y + abs_pivot.y),
455 (double)(+pos.x - abs_pivot.x))
456 - atan2((double)(-m_catchPos.y + abs_initial_pivot.y),
457 (double)(+m_catchPos.x - abs_initial_pivot.x));
458
459 newAngle = base::fmod_radians(newAngle);
460
461 // Is the "angle snap" is activated, we've to snap the angle
462 // to common (pixel art) angles.
463 if ((moveModifier & AngleSnapMovement) == AngleSnapMovement) {
464 // TODO make this configurable
465 static const double keyAngles[] = {
466 0.0, 26.565, 45.0, 63.435, 90.0, 116.565, 135.0, 153.435, 180.0,
467 180.0, -153.435, -135.0, -116, -90.0, -63.435, -45.0, -26.565
468 };
469
470 double newAngleDegrees = 180.0 * newAngle / PI3.14159265358979323846;
471
472 int closest = 0;
473 int last = sizeof(keyAngles) / sizeof(keyAngles[0]) - 1;
474 for (int i=0; i<=last; ++i) {
475 if (std::fabs(newAngleDegrees-keyAngles[closest]) >
476 std::fabs(newAngleDegrees-keyAngles[i]))
477 closest = i;
478 }
479
480 newAngle = PI3.14159265358979323846 * keyAngles[closest] / 180.0;
481 }
482
483 m_currentData.angle(newAngle);
484 }
485 break;
486
487 case PivotHandle:
488 {
489 // Calculate the new position of the pivot
490 gfx::PointF newPivot(m_initialData.pivot().x + (pos.x - m_catchPos.x),
491 m_initialData.pivot().y + (pos.y - m_catchPos.y));
492
493 m_currentData = m_initialData;
494 m_currentData.displacePivotTo(newPivot);
495 }
496 break;
497 }
498
499 if (updateBounds) {
500 m_currentData.bounds(bounds);
501 m_adjustPivot = true;
502 }
503
504 redrawExtraImage();
505
506 m_document->setTransformation(m_currentData);
507
508 // Get the new transformed corners
509 Transformation::Corners newCorners;
510 m_currentData.transformBox(newCorners);
511
512 // Create a union of all corners, and that will be the bounds to
513 // redraw of the sprite.
514 gfx::Rect fullBounds;
515 for (int i=0; i<Transformation::Corners::NUM_OF_CORNERS; ++i) {
516 fullBounds = fullBounds.createUnion(gfx::Rect((int)oldCorners[i].x, (int)oldCorners[i].y, 1, 1));
517 fullBounds = fullBounds.createUnion(gfx::Rect((int)newCorners[i].x, (int)newCorners[i].y, 1, 1));
518 }
519
520 // If "fullBounds" is empty is because the cel was not moved
521 if (!fullBounds.isEmpty()) {
522 // Notify the modified region.
523 m_document->notifySpritePixelsModified(
524 m_site.sprite(),
525 gfx::Region(fullBounds),
526 m_site.frame());
527 }
528}

863void PixelsMovement::drawParallelogram(
864 const Transformation& transformation,
865 doc::Image* dst, const doc::Image* src, const doc::Mask* mask,
866 const Transformation::Corners& corners,
867 const gfx::Point& leftTop)
868{
869 tools::RotationAlgorithm rotAlgo = Preferences::instance().selection.rotationAlgorithm();
870
871 // When the scale isn't modified and we have no rotation or a
872 // right/straight-angle, we should use the fast rotation algorithm,
873 // as it's pixel-perfect match with the original selection when just
874 // a translation is applied.
875 double angle = 180.0*transformation.angle()/PI3.14159265358979323846;
876 if (!Preferences::instance().selection.forceRotsprite() &&
877 (std::fabs(std::fmod(std::fabs(angle), 90.0)) < 0.01 ||
878 std::fabs(std::fmod(std::fabs(angle), 90.0)-90.0) < 0.01)) {
879 rotAlgo = tools::RotationAlgorithm::FAST;
880 }
881
882 // Don't use RotSprite if we are in "fast mode"
883 if (rotAlgo == tools::RotationAlgorithm::ROTSPRITE && m_fastMode) {
884 m_needsRotSpriteRedraw = true;
885 rotAlgo = tools::RotationAlgorithm::FAST;
886 }
887
888retry:; // In case that we don't have enough memory for RotSprite
889 // we can try with the fast algorithm anyway.
890
891 switch (rotAlgo) {
892
893 case tools::RotationAlgorithm::FAST:
894 doc::algorithm::parallelogram(
895 dst, src, (mask ? mask->bitmap(): nullptr),
896 int(corners.leftTop().x-leftTop.x),
897 int(corners.leftTop().y-leftTop.y),
898 int(corners.rightTop().x-leftTop.x),
899 int(corners.rightTop().y-leftTop.y),
900 int(corners.rightBottom().x-leftTop.x),
901 int(corners.rightBottom().y-leftTop.y),
902 int(corners.leftBottom().x-leftTop.x),
903 int(corners.leftBottom().y-leftTop.y));
904 break;
905
906 case tools::RotationAlgorithm::ROTSPRITE:
907 try {
908 doc::algorithm::rotsprite_image(
909 dst, src, (mask ? mask->bitmap(): nullptr),
910 int(corners.leftTop().x-leftTop.x),
911 int(corners.leftTop().y-leftTop.y),
912 int(corners.rightTop().x-leftTop.x),
913 int(corners.rightTop().y-leftTop.y),
914 int(corners.rightBottom().x-leftTop.x),
915 int(corners.rightBottom().y-leftTop.y),
916 int(corners.leftBottom().x-leftTop.x),
917 int(corners.leftBottom().y-leftTop.y));
918 }
919 catch (const std::bad_alloc&) {
920 StatusBar::instance()->showTip(1000,
921 "Not enough memory for RotSprite");
922
923 rotAlgo = tools::RotationAlgorithm::FAST;
924 goto retry;
925 }
926 break;

1036bool PixelsMovement::gotoFrame(const doc::frame_t deltaFrame)
1037{
1038 if (editMultipleCels()) {
1039 Layer* layer = m_site.layer();
1040 ASSERT(layer);
1041
1042 const doc::SelectedFrames frames = m_site.range().selectedFrames();
1043 doc::frame_t initialFrame = m_site.frame();
1044 doc::frame_t frame = initialFrame + deltaFrame;
1045
1046 if (frames.size() >= 2) {
1047 for (; !frames.contains(frame) &&
1048 !layer->cel(frame); frame+=deltaFrame) {
1049 if (deltaFrame > 0 && frame > frames.lastFrame()) {
1050 frame = frames.firstFrame();
1051 break;
1052 }
1053 else if (deltaFrame < 0 && frame < frames.firstFrame()) {
1054 frame = frames.lastFrame();
1055 break;
1056 }
1057 }
1058
1059 if (frame == initialFrame ||
1060 !frames.contains(frame) ||
1061 // TODO At the moment we don't support going to an empty cel,
1062 // so we don't handle these cases
1063 !layer->cel(frame)) {
1064 return false;
1065 }
1066
1067 // Rollback all the actions, go to the next/previous frame and
1068 // reproduce all transformation again so the new frame is the
1069 // preview for the transformation.
1070 m_tx.rollbackAndStartAgain();
1071
1072 {
1073 m_canHandleFrameChange = true;
1074 {
1075 ContextWriter writer(m_reader, 1000);
1076 writer.context()->setActiveFrame(frame);
1077 m_site.frame(frame);
1078 }
1079 m_canHandleFrameChange = false;
1080 }
1081
1082 reproduceAllTransformationsWithInnerCmds();
1083 return true;
1084 }
1085 }
1086 return false;
