3834void Timeline::updateDropRange(const gfx::Point& pt)
3835{
3836 const DropTarget oldDropTarget = m_dropTarget;
3837 m_dropTarget.hhit = DropTarget::HNone;
3838 m_dropTarget.vhit = DropTarget::VNone;
3839 m_dropTarget.outside = false;
3840
3841 if (m_state != STATE_MOVING_RANGE) {
3842 m_dropRange.clearRange();
3843 return;
3844 }
3845
3846 switch (m_range.type()) {
3847
3848 case Range::kCels:
3849 m_dropRange = m_range;
3850 m_dropRange.displace(m_hot.layer - m_clk.layer,
3851 m_hot.frame - m_clk.frame);
3852 break;
3853
3854 case Range::kFrames:
3855 case Range::kLayers:
3856 m_dropRange.clearRange();
3857 m_dropRange.startRange(m_rows[m_hot.layer].layer(), m_hot.frame, m_range.type());
3858 m_dropRange.endRange(m_rows[m_hot.layer].layer(), m_hot.frame);
3859 break;
3860 }
3861
3862 gfx::Rect bounds = getRangeBounds(m_dropRange);
3863
3864 if (pt.x < bounds.x + bounds.w/2) {
3865 m_dropTarget.hhit = DropTarget::Before;
3866 m_dropTarget.outside = (pt.x < bounds.x+2);
3867 }
3868 else {
3869 m_dropTarget.hhit = DropTarget::After;
3870 m_dropTarget.outside = (pt.x >= bounds.x2()-2);
3871 }
3872
3873 if (m_hot.veryBottom)
3874 m_dropTarget.vhit = DropTarget::VeryBottom;
3875 else if (pt.y < bounds.y + bounds.h/2)
3876 m_dropTarget.vhit = DropTarget::Top;
3877 // Special drop target for expanded groups
3878 else if (m_range.type() == Range::kLayers &&
3879 m_hot.layer >= 0 &&
3880 m_hot.layer < int(m_rows.size()) &&
3881 m_rows[m_hot.layer].layer()->isGroup() &&
3882 static_cast<LayerGroup*>(m_rows[m_hot.layer].layer())->isExpanded()) {
3883 m_dropTarget.vhit = DropTarget::FirstChild;
3884 }
3885 else {
3886 m_dropTarget.vhit = DropTarget::Bottom;
3887 }
