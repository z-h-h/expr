847bool AbstractCommand::checkIfConnectionEstablished(
848 const std::shared_ptr<SocketCore>& socket,
849 const std::string& connectedHostname, const std::string& connectedAddr,
850 uint16_t connectedPort)
851{
852 std::string error = socket->getSocketError();
853 if (error.empty()) {
854 return true;
855 }
856
857 // See also InitiateConnectionCommand::executeInternal()
858 e_->markBadIPAddress(connectedHostname, connectedAddr, connectedPort);
859 if (e_->findCachedIPAddress(connectedHostname, connectedPort).empty()) {
860 e_->removeCachedIPAddress(connectedHostname, connectedPort);
861 // Don't set error if proxy server is used and its method is GET.
862 if (resolveProxyMethod(req_->getProtocol()) != V_GET ||
863 !isProxyRequest(req_->getProtocol(), getOption())) {
864 e_->getRequestGroupMan()
865 ->getOrCreateServerStat(req_->getHost(), req_->getProtocol())
866 ->setError();
867 }
868 throw DL_RETRY_EX(fmt(MSG_ESTABLISHING_CONNECTION_FAILED, error.c_str()))DlRetryEx("AbstractCommand.cc", 868, fmt(gettext("Failed to establish connection, cause: %s"
), error.c_str()));
869 }
870
871 A2_LOG_INFO(fmt(MSG_CONNECT_FAILED_AND_RETRY, getCuid(),{ const std::shared_ptr<aria2::Logger>& logger = aria2
::LogFactory::getInstance(); if (logger->levelEnabled(Logger
::A2_INFO)) logger->log(Logger::A2_INFO, "AbstractCommand.cc"
, 872, fmt("CUID#%" "l" "d" " - Could not to connect to %s:%u. Trying another address"
, getCuid(), connectedAddr.c_str(), connectedPort)); }
872 connectedAddr.c_str(), connectedPort)){ const std::shared_ptr<aria2::Logger>& logger = aria2
::LogFactory::getInstance(); if (logger->levelEnabled(Logger
::A2_INFO)) logger->log(Logger::A2_INFO, "AbstractCommand.cc"
, 872, fmt("CUID#%" "l" "d" " - Could not to connect to %s:%u. Trying another address"
, getCuid(), connectedAddr.c_str(), connectedPort)); };
873 e_->setNoWait(true);
874 e_->addCommand(
875 InitiateConnectionCommandFactory::createInitiateConnectionCommand(
876 getCuid(), req_, fileEntry_, requestGroup_, e_));
877 return false;
878}
