174

193 &nxi,

197 &di,
198 x + i0 * d,
199 &di,
200 &zero,
201 ip_block.get(),

215void exhaustive_L2sqr_blas(
216 const float* x,
217 const float* y,
218 size_t d,
219 size_t nx,
220 size_t ny,

226

229 const size_t bs_y = distance_compute_blas_database_bs;

232 std::unique_ptr<float[]> x_norms(new float[nx]);
233 std::unique_ptr<float[]> del2;

235 fvec_norms_L2sqr(x_norms.get(), x, d, nx);
236
237 if (!y_norms) {
238 float* y_norms2 = new float[ny];
239 del2.reset(y_norms2);
240 fvec_norms_L2sqr(y_norms2, y, d, ny);
241 y_norms = y_norms2;
242 }
243
244 for (size_t i0 = 0; i0 < nx; i0 += bs_x) {
245 size_t i1 = i0 + bs_x;
246 if (i1 > nx)
247 i1 = nx;
248
249 res.begin_multiple(i0, i1);
250
251 for (size_t j0 = 0; j0 < ny; j0 += bs_y) {
252 size_t j1 = j0 + bs_y;
253 if (j1 > ny)
254 j1 = ny;
255 /* compute the actual dot products */
256 {
257 float one = 1, zero = 0;
258 FINTEGERint nyi = j1 - j0, nxi = i1 - i0, di = d;
259 sgemm_("Transpose",
260 "Not transpose",
261 &nyi,
262 &nxi,
263 &di,
264 &one,
265 y + j0 * d,
266 &di,
267 x + i0 * d,
268 &di,
269 &zero,
270 ip_block.get(),
271 &nyi);
272 }
273#pragma omp parallel for
274 for (int64_t i = i0; i < i1; i++) {
275 float* ip_line = ip_block.get() + (i - i0) * (j1 - j0);
276
277 for (size_t j = j0; j < j1; j++) {
278 float ip = *ip_line;

284 dis = 0;

287 ip_line++;
288 }

291 }
292 res.end_multiple();
293 InterruptCallback::check();
294 }
