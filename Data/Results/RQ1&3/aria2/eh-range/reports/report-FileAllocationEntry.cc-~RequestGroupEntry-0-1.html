779void RequestGroup::tryAutoFileRenaming()
780{
781 if (!option_->getAsBool(PREF_AUTO_FILE_RENAMING)) {
782 throw DOWNLOAD_FAILURE_EXCEPTION2(
783 fmt(MSG_FILE_ALREADY_EXISTS, getFirstFilePath().c_str()),
784 error_code::FILE_ALREADY_EXISTS);
785 }
786
787 std::string filepath = getFirstFilePath();
788 if (filepath.empty()) {
789 throw DOWNLOAD_FAILURE_EXCEPTION2(
790 fmt("File renaming failed: %s", getFirstFilePath().c_str()),
791 error_code::FILE_RENAMING_FAILED);
792 }
793 auto fn = filepath;
794 std::string ext;
795 const auto idx = fn.find_last_of(".");
796 const auto slash = fn.find_last_of("\\/");
797 // Do extract the extension, as in "file.ext" = "file" and ".ext",
798 // but do not consider ".file" to be a file name without extension instead
799 // of a blank file name and an extension of ".file"
800 if (idx != std::string::npos &&
801 // fn has no path component and starts with a dot, but has no extension
802 // otherwise
803 idx != 0 &&
804 // has a file path component if we found a slash.
805 // if slash == idx - 1 this means a form of "*/.*", so the file name
806 // starts with a dot, has no extension otherwise, and therefore do not
807 // extract an extension either
808 (slash == std::string::npos || slash < idx - 1)) {
809 ext = fn.substr(idx);
810 fn = fn.substr(0, idx);
811 }
812 for (int i = 1; i < 10000; ++i) {
813 auto newfilename = fmt("%s.%d%s", fn.c_str(), i, ext.c_str());
814 File newfile(newfilename);
815 File ctrlfile(newfile.getPath() + DefaultBtProgressInfoFile::getSuffix());
816 if (!newfile.exists() || (newfile.exists() && ctrlfile.exists())) {
817 downloadContext_->getFirstFileEntry()->setPath(newfile.getPath());
818 return;
819 }
820 }
821 throw DOWNLOAD_FAILURE_EXCEPTION2(
822 fmt("File renaming failed: %s", getFirstFilePath().c_str()),
823 error_code::FILE_RENAMING_FAILED);
824}
