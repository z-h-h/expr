248///
249/// Invokes t; if exceptions are enabled (if not compiled with -fno-exceptions),
250/// catches a thrown exception of any type and invokes c, forwarding any
251/// trailing arguments.

255/// inlined native code at the call-site.
256///

258///
259/// int input = // ...
260/// int def = 45;
261/// auto result = catch_exception(
262/// [=] {
263/// if (input < 0) throw 11;
264/// return input;
265/// },
266/// [](int num) { return num; },
267/// def);

275 decltype(FOLLY_DECLVAL(Catch &&)static_cast<Catch && (*)() noexcept>(nullptr)()(FOLLY_DECLVAL(CatchA &&)static_cast<CatchA && (*)() noexcept>(nullptr)(
)...))>>
276FOLLY_ERASE_TRYCATCHinline __attribute__((__always_inline__)) __attribute__((__visibility__
("hidden"))) R catch_exception(Try&& t, Catch&& c, CatchA&&... a) {
277#if FOLLY_HAS_EXCEPTIONS1

329// exception_ptr_get_object
330//
331// Returns the address of the stored exception as if it were upcast to the
332// given type, if it could be upcast to that type.
333template <typename T>
334T* exception_ptr_get_object(std::exception_ptr const& ptr) noexcept {
335 static_assert(!std::is_reference<T>::value, "is a reference");
336 auto target = type_info_of<T>();
337 auto object = !target ? nullptr : exception_ptr_get_object(ptr, target);
338 return static_cast<T*>(object);
