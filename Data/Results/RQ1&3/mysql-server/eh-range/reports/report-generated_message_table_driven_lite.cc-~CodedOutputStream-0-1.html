902 static_assert(std::is_unsigned<T>::value,
903 "Varint serialization must be unsigned");

905 ptr[0] = static_cast<uint8>(value);
906 return ptr + 1;

909 value >>= 7;

912 return ptr + 2;

914 ptr++;

917 value >>= 7;
918 ++ptr;
919 } while (PROTOBUF_PREDICT_FALSE(value >= 0x80));

1383inline std::pair<uint32, bool> CodedInputStream::ReadTagWithCutoffNoLastTag(
1384 uint32 cutoff) {
1385 // In performance-sensitive code we can expect cutoff to be a compile-time
1386 // constant, and things like "cutoff >= kMax1ByteVarint" to be evaluated at
1387 // compile time.
1388 uint32 first_byte_or_zero = 0;
1389 if (PROTOBUF_PREDICT_TRUE(buffer_ < buffer_end_)) {
1390 // Hot case: buffer_ non_empty, buffer_[0] in [1, 128).
1391 // TODO(gpike): Is it worth rearranging this? E.g., if the number of fields
1392 // is large enough then is it better to check for the two-byte case first?
1393 first_byte_or_zero = buffer_[0];
1394 if (static_cast<int8>(buffer_[0]) > 0) {
1395 const uint32 kMax1ByteVarint = 0x7f;
1396 uint32 tag = buffer_[0];
1397 Advance(1);
1398 return std::make_pair(tag, cutoff >= kMax1ByteVarint || tag <= cutoff);
1399 }
1400 // Other hot case: cutoff >= 0x80, buffer_ has at least two bytes available,
1401 // and tag is two bytes. The latter is tested by bitwise-and-not of the
1402 // first byte and the second byte.
1403 if (cutoff >= 0x80 && PROTOBUF_PREDICT_TRUE(buffer_ + 1 < buffer_end_) &&
1404 PROTOBUF_PREDICT_TRUE((buffer_[0] & ~buffer_[1]) >= 0x80)) {
1405 const uint32 kMax2ByteVarint = (0x7f << 7) + 0x7f;
1406 uint32 tag = (1u << 7) * buffer_[1] + (buffer_[0] - 0x80);
1407 Advance(2);
1408 // It might make sense to test for tag == 0 now, but it is so rare that
1409 // that we don't bother. A varint-encoded 0 should be one byte unless
1410 // the encoder lost its mind. The second part of the return value of
1411 // this function is allowed to be either true or false if the tag is 0,
1412 // so we don't have to check for tag == 0. We may need to check whether
1413 // it exceeds cutoff.
1414 bool at_or_below_cutoff = cutoff >= kMax2ByteVarint || tag <= cutoff;
1415 return std::make_pair(tag, at_or_below_cutoff);
1416 }
1417 }
1418 // Slow path
1419 const uint32 tag = ReadTagFallback(first_byte_or_zero);
1420 return std::make_pair(tag, static_cast<uint32>(tag - 1) < cutoff);
1421}
