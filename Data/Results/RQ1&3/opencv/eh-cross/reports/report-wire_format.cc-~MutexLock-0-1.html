84bool WireFormat::SkipField(io::CodedInputStream* input, uint32 tag,
85 UnknownFieldSet* unknown_fields) {
86 int number = WireFormatLite::GetTagFieldNumber(tag);
87 // Field number 0 is illegal.
88 if (number == 0) return false;
89
90 switch (WireFormatLite::GetTagWireType(tag)) {
91 case WireFormatLite::WIRETYPE_VARINT: {
92 uint64 value;
93 if (!input->ReadVarint64(&value)) return false;
94 if (unknown_fields != NULL__null) unknown_fields->AddVarint(number, value);
95 return true;
96 }
97 case WireFormatLite::WIRETYPE_FIXED64: {
98 uint64 value;
99 if (!input->ReadLittleEndian64(&value)) return false;
100 if (unknown_fields != NULL__null) unknown_fields->AddFixed64(number, value);
101 return true;
102 }
103 case WireFormatLite::WIRETYPE_LENGTH_DELIMITED: {
104 uint32 length;
105 if (!input->ReadVarint32(&length)) return false;
106 if (unknown_fields == NULL__null) {
107 if (!input->Skip(length)) return false;
108 } else {
109 if (!input->ReadString(unknown_fields->AddLengthDelimited(number),
110 length)) {
111 return false;
112 }
113 }
114 return true;
115 }
116 case WireFormatLite::WIRETYPE_START_GROUP: {
117 if (!input->IncrementRecursionDepth()) return false;
118 if (!SkipMessage(input, (unknown_fields == NULL__null) ?
119 NULL__null : unknown_fields->AddGroup(number))) {
120 return false;
121 }
122 input->DecrementRecursionDepth();
123 // Check that the ending tag matched the starting tag.
124 if (!input->LastTagWas(WireFormatLite::MakeTag(
125 WireFormatLite::GetTagFieldNumber(tag),
126 WireFormatLite::WIRETYPE_END_GROUP))) {
127 return false;
128 }
129 return true;
130 }
131 case WireFormatLite::WIRETYPE_END_GROUP: {
132 return false;
133 }
134 case WireFormatLite::WIRETYPE_FIXED32: {
135 uint32 value;
136 if (!input->ReadLittleEndian32(&value)) return false;
137 if (unknown_fields != NULL__null) unknown_fields->AddFixed32(number, value);
138 return true;
139 }
140 default: {
141 return false;
142 }
143 }

719bool WireFormat::ParseAndMergeMessageSetItem(
720 io::CodedInputStream* input,
721 Message* message) {
722 const Reflection* message_reflection = message->GetReflection();
723
724 // This method parses a group which should contain two fields:
725 // required int32 type_id = 2;
726 // required data message = 3;
727
728 uint32 last_type_id = 0;
729
730 // Once we see a type_id, we'll look up the FieldDescriptor for the
731 // extension.
732 const FieldDescriptor* field = NULL__null;
733
734 // If we see message data before the type_id, we'll append it to this so
735 // we can parse it later.
736 string message_data;
737
738 while (true) {
739 uint32 tag = input->ReadTag();
740 if (tag == 0) return false;
741
742 switch (tag) {
743 case WireFormatLite::kMessageSetTypeIdTag: {
744 uint32 type_id;
745 if (!input->ReadVarint32(&type_id)) return false;
746 last_type_id = type_id;
747 field = message_reflection->FindKnownExtensionByNumber(type_id);
748
749 if (!message_data.empty()) {
750 // We saw some message data before the type_id. Have to parse it
751 // now.
752 io::ArrayInputStream raw_input(message_data.data(),
753 message_data.size());
754 io::CodedInputStream sub_input(&raw_input);
755 if (!ParseAndMergeMessageSetField(last_type_id, field, message,
756 &sub_input)) {
757 return false;
758 }
759 message_data.clear();
760 }
761
762 break;
763 }
764
765 case WireFormatLite::kMessageSetMessageTag: {
766 if (last_type_id == 0) {
767 // We haven't seen a type_id yet. Append this data to message_data.
768 string temp;
769 uint32 length;
770 if (!input->ReadVarint32(&length)) return false;
771 if (!input->ReadString(&temp, length)) return false;
772 io::StringOutputStream output_stream(&message_data);
773 io::CodedOutputStream coded_output(&output_stream);
774 coded_output.WriteVarint32(length);
775 coded_output.WriteString(temp);
776 } else {
777 // Already saw type_id, so we can parse this directly.
778 if (!ParseAndMergeMessageSetField(last_type_id, field, message,
779 input)) {
780 return false;
781 }
782 }
783
784 break;
785 }
786
787 case WireFormatLite::kMessageSetItemEndTag: {
788 return true;
789 }
790
791 default: {
792 if (!SkipField(input, tag, NULL__null)) return false;
793 }
794 }
795 }
