323bool WatcherRunner::watch(const PlatformProcess& child) const {
324 int process_status = 0;

332 // Worker does not exist or never existed.

338 if (!status.ok() && getUnixTime() >= delayedTime()) {
339 // Since the watchdog cannot use the logger plugin the error message
340 // should be logged to stderr and to the system log.
